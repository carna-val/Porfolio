/** * === CANVAS PARTICULES === */ const canvas = document.getElementById('space-canvas'); const ctx = canvas.getContext('2d'); let width, height; let particles = []; let mouse = { x: 0, y: 0 }; function initCanvas() { width = window.innerWidth; height = window.innerHeight; canvas.width = width; canvas.height = height; createParticles(); } function createParticles() { particles = []; const count = (width * height) / 8000; for(let i = 0; i < count; i++) { particles.push({ x: Math.random() * width, y: Math.random() * height, vx: (Math.random() - 0.5) * 0.3, vy: (Math.random() - 0.5) * 0.3, size: Math.random() * 2.5, color: Math.random() > 0.85 ? '#bc13fe' : '#ffffff', opacity: Math.random() * 0.5 + 0.5 }); } } function animateCanvas() { ctx.clearRect(0, 0, width, height); for(let i = 0; i < particles.length; i++) { let p = particles[i]; // Interaction souris const dx = mouse.x - p.x; const dy = mouse.y - p.y; const dist = Math.sqrt(dx * dx + dy * dy); if(dist < 150) { const force = (150 - dist) / 150; p.vx -= (dx / dist) * force * 0.2; p.vy -= (dy / dist) * force * 0.2; } p.x += p.vx; p.y += p.vy; // Friction p.vx *= 0.99; p.vy *= 0.99; if(p.x < 0 || p.x > width) p.vx *= -1; if(p.y < 0 || p.y > height) p.vy *= -1; // Dessiner Ã©toile avec glow ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); ctx.fillStyle = p.color; ctx.globalAlpha = p.opacity; ctx.fill(); // Glow effect ctx.shadowBlur = 10; ctx.shadowColor = p.color; ctx.fill(); ctx.shadowBlur = 0; ctx.globalAlpha = 1; // Constellations for(let j = i + 1; j < particles.length; j++) { let p2 = particles[j]; let distance = Math.hypot(p.x - p2.x, p.y - p2.y); if(distance < 120) { ctx.beginPath(); ctx.strokeStyle = `rgba(0, 243, 255, ${0.15 - distance/800})`; ctx.lineWidth = 0.5; ctx.moveTo(p.x, p.y); ctx.lineTo(p2.x, p2.y); ctx.stroke(); } } } requestAnimationFrame(animateCanvas); }
